def clear_screen():
    os.system('cls' if os.name == 'nt' else 'clear')

def type_text(text, speed=0.015):
    """몰입감을 위한 타이핑 연출 효과"""
    for char in text:
        sys.stdout.write(char)
        sys.stdout.flush()
        time.sleep(speed)
    print()

def print_header(title):
    print("\n" + "="*60)
    print(f" {title} ".center(60, '='))
    print("="*60 + "\n")

class Player:
    def __init__(self):
        self.hp = 1500
        self.max_hp = 1500
        self.hunger = 100
        self.thirst = 100
        self.max_ram = 4
        self.materials = 0
        
        # 전직 가중치
        self.weights = {"kinetic": 0, "scrap": 0, "cyber": 0}
        
        # 인벤토리 및 장비
        self.inventory = []
        self.equipment = {"weapon": "WEAPON_NONE", "armor": None, "accessory": None}

    def get_attack_power(self):
        return ITEMS_DB[self.equipment["weapon"]]["power"]

    def consume_resources(self):
        """이동 시 이중 자원 루프 감산 공식 적용"""
        self.hunger = max(0, self.hunger - 5)
        self.thirst = max(0, self.thirst - 6)
        
        if self.hunger == 0 or self.thirst == 0:
            self.hp -= 50
            print(" [GLITCH ALERT] 신체 연료 고갈! 괴사가 시작됩니다. (HP -50)")
            time.sleep(1)
            if self.hp <= 0:
                print("\n[SYSTEM] 신체 신호 영구 정지. 불량 코드가 완전히 소거되었습니다.")
                sys.exit()

    def show_status(self):
        print("\n[내 의체 상태창]")
        print(f" - HP: {self.hp}/{self.max_hp} | 허기: {self.hunger}/100 | 갈증: {self.thirst}/100")
        wpn_name = ITEMS_DB[self.equipment['weapon']]['name']
        print(f" - 장착 무기: {wpn_name} (위력: {self.get_attack_power()}) | 가용 RAM: {self.max_ram}")
        print(f" - 가중치 배열 -> [컴뱃:{self.weights['kinetic']}] [메카:{self.weights['scrap']}] [넷:{self.weights['cyber']}]")

    def manage_inventory(self):
        """플레이 타임을 증가시키는 인벤토리 정비 시스템"""
        clear_screen()
        print_header("INVENTORY (정비)")
        if not self.inventory:
            print("인벤토리가 비어 있습니다.")
        else:
            for i, item_id in enumerate(self.inventory):
                print(f" [{i+1}] {ITEMS_DB[item_id]['name']} (위력: {ITEMS_DB[item_id]['power']})")
            
            choice = input("\n장착할 아이템 번호를 선택하세요 (취소: 0): ")
            if choice.isdigit() and 0 < int(choice) <= len(self.inventory):
                item = self.inventory[int(choice)-1]
                self.equipment["weapon"] = item # 간이화하여 모두 무기로 장착되게 처리
                print(f" -> {ITEMS_DB[item]['name']} 을(를) 시스템 소켓에 결속했습니다.")
                time.sleep(1)


class GameMap:
    def __init__(self):
        self.size = 5
        self.player_pos = [0, 0]
        self.bunker_pos = [4, 4]
        # 맵 곳곳에 5개의 세션을 무작위로 흩뿌림 (벙커 4,4와 시작점 0,0 제외)
        coords = [(x, y) for x in range(5) for y in range(5) if (x,y) not in [(0,0), (4,4)]]
        self.event_locations = random.sample(coords, 4) # 1~4세션 위치
        self.session_index = 0

    def draw(self):
        print("\n [쓰레기 바다 그리드 스캐너]")
        for y in range(self.size - 1, -1, -1):
            row_str = ""
            for x in range(self.size):
                if [x, y] == self.player_pos:
                    row_str += "[ P ] "
                elif [x, y] == self.bunker_pos:
                    row_str += "[ ☗ ] "
                else:
                    row_str += "[ · ] "
            print("  " + row_str)
        print(" (P: 플레이어 위치 | ☗: 벙커 목표지점)\n")

def combat_loop(player, is_boss=False):
    clear_screen()
    if is_boss:
        name, hp, e_def, atk = "스캐브 컬렉터 [BOSS]", 35000, 45, 180
        print_header("WARNING: 숙청 시퀀스 가동")
        type_text("대형 분쇄 기계 '스캐브 컬렉터'가 벙커를 향해 돌진해옵니다!!", 0.03)
    else:
        name, hp, e_def, atk = "오염된 스캐브 드론", 8000, 5, 80
        print_header("RANDOM ENCOUNTER")
        type_text(f"쓰레기 더미에서 {name}이 튀어나왔습니다!", 0.02)

    turn = 1
    learning_index = 0 # 딥러닝 패턴 노출도 (E)
    consecutive_attacks = 0

    while hp > 0 and player.hp > 0:
        if is_boss and turn > 15:
            type_text("\n[ALERT] 15턴 임계점 초과! 거점이 고철 분진으로 분쇄되었습니다. GAME OVER.")
            sys.exit()

        print(f"\n--- 턴 [{turn}] | {name} HP: {hp} ---")
        if is_boss:
            print(f" [시스템] 보스 패턴 분석 지수 (E): {learning_index}/10")
        
        print(" 1. 주무기 타격 (ATTACK)")
        print(" 2. 급조 바리케이드 (DEFENSE - 분석 지수 차감)")
        print(" 3. 패킷 우회 교란 (BYPASS - RAM 2 소모, 분석 지수 초기화)")
        
        cmd = input("명령 코드 (1-3): ")
        
        # 1. 플레이어 행동
        if cmd == "1":
            consecutive_attacks += 1
            if consecutive_attacks >= 2 and is_boss:
                learning_index += 3
                print(" -> [WARNING] 동일 공격 반복! 보스가 궤적을 딥러닝 중입니다. (E +3)")
            
            # 패턴 노출 페널티(P) 계산
            penalty = 1.0
            if learning_index > 10:
                penalty = max(0.5, 1.0 - (learning_index - 10) * 0.05)
            
            dmg = math.floor((player.get_attack_power() * penalty) * 100) - e_def
            dmg = max(100, dmg + random.randint(-50, 50))
            hp -= dmg
            print(f" ⚔️ 적에게 {dmg} 의 대미지를 꽂아넣었습니다!")

        elif cmd == "2":
            consecutive_attacks = 0
            learning_index = max(0, learning_index - 4)
            atk = int(atk * 0.5) # 이번 턴 피해 반감
            print(" 🛡️ 바리케이드 전개! 적의 분석 궤적을 방해하고 방어력을 올립니다.")

        elif cmd == "3":
            if player.max_ram >= 2:
                consecutive_attacks = 0
                learning_index = 0
                print(" 📡 가상 넷 오버레이 신호 방출! 보스의 센서를 완전 초기화했습니다.")
            else:
                print(" RAM 가용량이 부족합니다! 행동 낭비!")
        else:
            print(" 잘못된 커맨드. 턴을 낭비했습니다.")

        # 2. 적 반격
        if hp > 0:
            player.hp -= atk
            print(f" 💥 {name}의 반격! 의체가 {atk} 의 손상을 입었습니다.")
            if cmd == "2": atk = atk * 2 # 반감 복구

        time.sleep(1.5)
        turn += 1

    if player.hp <= 0:
        type_text("\n신체 손상 100%. 불량 코드가 폐기되었습니다. GAME OVER.")
        sys.exit()
    
    type_text(f"\n✨ {name} 제압 완료!", 0.02)
    player.materials += 15
    time.sleep(1)

def run_game():
    clear_screen()
    print_header("1막: 낙인 (Protocol: Stigma)")
    type_text("[SYSTEM]: 생체 신호 복구 중... 인코딩 결함 발견.", 0.03)
    type_text("당신은 '네오 아크'의 실험실에서 불량품으로 폐기되어 데드존의 쓰레기 바다에 버려졌습니다.", 0.03)
    type_text("[WARNING]: 무국적 불량 코드(Civilian) 모드 작동.\n", 0.03)
    time.sleep(1)

    player = Player()
    grid = GameMap()

    while True:
        clear_screen()
        grid.draw()
        player.show_status()
        
        print("\n [액션 커맨드]")
        print(" W(북), A(서), S(남), D(동) 이동 | I(인벤토리 정비)")
        move = input("입력: ").strip().upper()

        if move == "I":
            player.manage_inventory()
            continue
        
        # 이동 처리
        px, py = grid.player_pos[0], grid.player_pos[1]
        if move == "W" and py < grid.size - 1: py += 1
        elif move == "S" and py > 0: py -= 1
        elif move == "A" and px > 0: px -= 1
        elif move == "D" and px < grid.size - 1: px += 1
        else:
            print("이동할 수 없는 장벽입니다.")
            time.sleep(0.5)
            continue

        grid.player_pos = [px, py]
        player.consume_resources()

        # 타일 이벤트 판정
        current_loc = (px, py)
        if current_loc == tuple(grid.bunker_pos):
            # 벙커 도착 (세션 5 강제 실행 및 보스전)
            handle_session(player, SESSIONS_DB[4])
            combat_loop(player, is_boss=True)
            run_ending(player)
            break
        elif current_loc in grid.event_locations:
            # 주요 세션 이벤트 트리거
            session_data = SESSIONS_DB[grid.session_index]
            handle_session(player, session_data)
            grid.event_locations.remove(current_loc)
            grid.session_index += 1
        else:
            # 30% 확률로 일반 몬스터와 조우 (시간을 소모시키는 핵심 루프)
            if random.random() < 0.3:
                combat_loop(player, is_boss=False)

def handle_session(player, session):
    clear_screen()
    print_header(f"EVENT: {session['title']}")
    type_text(session['text'], 0.02)
    print()
    
    for i, choice in enumerate(session['choices']):
        print(f" [{i+1}] {choice['text']}")
    
    while True:
        ans = input("\n당신의 행동을 선택하십시오 (1-3): ")
        if ans in ["1", "2", "3"]:
            idx = int(ans) - 1
            choice_data = session['choices'][idx]
            
            # 가중치 누적
            player.weights[choice_data['weight']] += 1
            
            # 보상 처리
            if "reward" in choice_data and choice_data["reward"]:
                if choice_data["reward"] == "SCRAP_MAT":
                    player.materials += 30

            print(f"\n -> {choice_data['log']}")
            time.sleep(2)
            break

1. **시각적 그리드 탐색 엔진 (`GameMap.draw()`)**

3. **랜덤 조우 확률과 자원 압박의 줄다리기**
